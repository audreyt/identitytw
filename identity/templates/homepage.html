ID_REGEX = /^[A-Z]{1}[A-Z0-9]{1}[0-9]{8}$/;
ID_LETTERS = 'ABCDEFGHJKLMNPQRSTUVXYWZIO';

function isValid(id) {
  id = id.toUpperCase();
  if (!ID_REGEX.test(id)) {
    return false;
  }
  
  let letterIndex = ID_LETTERS.indexOf(id[0]);
  let sum = Math.floor(letterIndex / 10 + 1) + (letterIndex * 9);

  // Legacy ARC 
  if ((/[A-Z]/).test(id[1])) {
    sum += (ID_LETTERS.indexOf(id[1]) % 10) * 8;
  } else {
    sum += parseInt(id[1]) * 8;
  }

  for (i = 2 ; i < 9 ; i++) {
    sum += (9 - i) * parseInt(id[i]);
  }
  sum += parseInt(id[9]);

  return sum % 10 === 0;
}
